#include <bits/stdc++.h> 
using namespace std; 
#define ull unsigned long long int 
  
 
void kFactors(ull n, ull k) 
{ 

    vector<int> P; 
  

    while (n%2 == 0) 
    { 
        P.push_back(2); 
        n /= 2; 
    } 
  
    for (int i=3; i*i<=n; i=i+2) 
    { 
        while (n%i == 0) 
        { 
            n = n/i; 
            P.push_back(i); 
        } 
    }
    if (n > 2) 
        P.push_back(n); 
    if (P.size() < k) 
    { 
        cout << "0" << endl; 
        return; 
    } 
    else cout<<"1"<<endl;
  

} 
  

int main() 
{ ull t;
  cin>>t;
  while(t--)
  {
    ull n, k;
    cin>>n>>k;
    
    kFactors(n, k); 
  }
  
}
